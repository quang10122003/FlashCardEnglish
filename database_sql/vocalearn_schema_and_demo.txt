-- =========================================
-- VOCALearn Demo SQL (MySQL 8+)
-- Tao database + schema theo entity + du lieu demo
-- =========================================

DROP DATABASE IF EXISTS `vocalearn_demo_db`;
CREATE DATABASE `vocalearn_demo_db`
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;
USE `vocalearn_demo_db`;

-- =========================================
-- 1) TABLE DEFINITIONS
-- =========================================

CREATE TABLE `user` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `fullName` VARCHAR(100) NOT NULL,
  `birthday` DATE NULL,
  `email` VARCHAR(100) NOT NULL UNIQUE,
  `avatar` VARCHAR(255) NULL,
  `phoneNumber` VARCHAR(50) NULL,
  `address` VARCHAR(255) NULL,
  `accountName` VARCHAR(50) NOT NULL UNIQUE,
  `passWord` VARCHAR(255) NOT NULL,
  `createAt` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `isDeleted` TINYINT(1) NOT NULL DEFAULT 0,
  `role` ENUM('USER','ADMIN') NOT NULL DEFAULT 'USER'
) ENGINE=InnoDB;

CREATE TABLE `exam_type` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `type` VARCHAR(255) NOT NULL,
  `isDeleted` TINYINT(1) NOT NULL DEFAULT 0
) ENGINE=InnoDB;

CREATE TABLE `exam_collection` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `collection` VARCHAR(255) NOT NULL,
  `isDeleted` TINYINT(1) NOT NULL DEFAULT 0
) ENGINE=InnoDB;

CREATE TABLE `flashCard_topic` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `title` VARCHAR(255) NOT NULL,
  `visit_count` INT NOT NULL DEFAULT 0,
  `createAt` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `userID` INT NOT NULL,
  `status` ENUM('PRIVATE','PUBLIC') DEFAULT 'PUBLIC',
  `status_gain` TINYINT(1) NULL,
  `learning_status` ENUM('MASTERED','IN_PROGRESS','NEW','REVIEW_NEEDED') DEFAULT 'NEW',
  CONSTRAINT `fk_flashcard_topic_user` FOREIGN KEY (`userID`) REFERENCES `user`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `flashCard` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `title` VARCHAR(100) NOT NULL,
  `reviewDate` DATE NULL,
  `cycle` INT NOT NULL DEFAULT 1,
  `learningStatus` ENUM('MASTERED','IN_PROGRESS','NEW','REVIEW_NEEDED') NOT NULL DEFAULT 'NEW',
  `createdAt` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `topicID` INT NOT NULL,
  CONSTRAINT `fk_flashcard_topic` FOREIGN KEY (`topicID`) REFERENCES `flashCard_topic`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `card` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `terminology` VARCHAR(255) NOT NULL,
  `hint` TEXT NOT NULL,
  `definition` TEXT NULL,
  `image` VARCHAR(255) NULL,
  `audio` VARCHAR(255) NULL,
  `pronounce` VARCHAR(255) NULL,
  `level` INT DEFAULT 1,
  `partOfSpeech` VARCHAR(50) NOT NULL,
  `example` TEXT NULL,
  `isRemember` INT DEFAULT 0,
  `createAt` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `flashCardID` INT NOT NULL,
  CONSTRAINT `fk_card_flashcard` FOREIGN KEY (`flashCardID`) REFERENCES `flashCard`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `blog_category` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `title` VARCHAR(255) NOT NULL
) ENGINE=InnoDB;

CREATE TABLE `blog` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `title` VARCHAR(255) NOT NULL,
  `category_id` INT NOT NULL,
  `short_detail` TEXT NOT NULL,
  `image` VARCHAR(255) NULL,
  `detail` LONGTEXT NOT NULL,
  `createAt` DATETIME NULL,
  `author` VARCHAR(255) NULL,
  CONSTRAINT `fk_blog_category` FOREIGN KEY (`category_id`) REFERENCES `blog_category`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `exam` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `duration` INT NULL,
  `title` VARCHAR(100) NOT NULL,
  `year` INT NULL,
  `type_id` INT NULL,
  `collection_id` INT NULL,
  `attemps` INT NOT NULL DEFAULT 0,
  `isRandom` TINYINT(1) NOT NULL DEFAULT 0,
  `createdAt` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `isDeleted` TINYINT(1) NOT NULL DEFAULT 0,
  `fileImportName` VARCHAR(255) NULL,
  CONSTRAINT `fk_exam_type` FOREIGN KEY (`type_id`) REFERENCES `exam_type`(`id`),
  CONSTRAINT `fk_exam_collection` FOREIGN KEY (`collection_id`) REFERENCES `exam_collection`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `group_question` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `part` VARCHAR(20) NOT NULL,
  `title` VARCHAR(255) NULL,
  `content` TEXT NULL,
  `is_contribute` TINYINT(1) DEFAULT 0,
  `bank_group_id` BIGINT NULL,
  `questionRange` VARCHAR(50) NULL,
  `exam_id` INT NOT NULL,
  CONSTRAINT `fk_group_question_exam` FOREIGN KEY (`exam_id`) REFERENCES `exam`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `group_question_image` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `url` VARCHAR(255) NOT NULL,
  `group_id` INT NOT NULL,
  CONSTRAINT `fk_group_image_group` FOREIGN KEY (`group_id`) REFERENCES `group_question`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `group_question_audio` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `url` VARCHAR(255) NOT NULL,
  `group_id` INT NOT NULL,
  CONSTRAINT `fk_group_audio_group` FOREIGN KEY (`group_id`) REFERENCES `group_question`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `toeicQuestion` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `indexNumber` INT NULL,
  `part` VARCHAR(50) NULL,
  `detail` TEXT NULL,
  `result` VARCHAR(1) NOT NULL,
  `clarify` TEXT NULL,
  `audio` VARCHAR(255) NULL,
  `conversation` VARCHAR(255) NULL,
  `is_contribute` TINYINT(1) DEFAULT 0,
  `bank_question_id` INT NULL,
  `createAt` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `examID` INT NOT NULL,
  `group_question_id` INT NULL,
  CONSTRAINT `fk_toeic_exam` FOREIGN KEY (`examID`) REFERENCES `exam`(`id`),
  CONSTRAINT `fk_toeic_group` FOREIGN KEY (`group_question_id`) REFERENCES `group_question`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `toeicQuestionImage` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `url` VARCHAR(255) NOT NULL,
  `createdAt` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `toeicQuestionID` INT NOT NULL,
  CONSTRAINT `fk_toeic_image_question` FOREIGN KEY (`toeicQuestionID`) REFERENCES `toeicQuestion`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `toeicQuestionOption` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `detail` TEXT NOT NULL,
  `mark` VARCHAR(10) NULL,
  `createAt` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `toeicQuestionID` INT NOT NULL,
  CONSTRAINT `fk_toeic_option_question` FOREIGN KEY (`toeicQuestionID`) REFERENCES `toeicQuestion`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `evaluate` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `userID` INT NOT NULL,
  `content` TEXT NULL,
  `star` INT NOT NULL,
  `image` VARCHAR(255) NULL,
  `admin_reply` TEXT NULL,
  `replyAt` DATETIME NULL,
  `createAt` DATETIME DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT `fk_evaluate_user` FOREIGN KEY (`userID`) REFERENCES `user`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `comment` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `userID` INT NULL,
  `examID` INT NOT NULL,
  `content` TEXT NOT NULL,
  `createAt` DATETIME DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT `fk_comment_user` FOREIGN KEY (`userID`) REFERENCES `user`(`id`),
  CONSTRAINT `fk_comment_exam` FOREIGN KEY (`examID`) REFERENCES `exam`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `comment_replies` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `commentID` INT NOT NULL,
  `userID` INT NOT NULL,
  `parentReplyID` INT NULL,
  `content` TEXT NOT NULL,
  `createAt` DATETIME DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT `fk_reply_comment` FOREIGN KEY (`commentID`) REFERENCES `comment`(`id`),
  CONSTRAINT `fk_reply_user` FOREIGN KEY (`userID`) REFERENCES `user`(`id`),
  CONSTRAINT `fk_reply_parent` FOREIGN KEY (`parentReplyID`) REFERENCES `comment_replies`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `examReview` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `userID` INT NOT NULL,
  `examID` INT NOT NULL,
  `createAt` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `duration` INT NOT NULL,
  `selectedPart` VARCHAR(255) NOT NULL,
  `result` INT NOT NULL,
  `incorrect` INT NULL,
  CONSTRAINT `fk_examreview_user` FOREIGN KEY (`userID`) REFERENCES `user`(`id`),
  CONSTRAINT `fk_examreview_exam` FOREIGN KEY (`examID`) REFERENCES `exam`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `questionReview` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `examReviewID` INT NOT NULL,
  `toeicQuestionID` INT NOT NULL,
  `userAnswer` VARCHAR(1) NULL,
  `createAt` DATETIME DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT `fk_qreview_examreview` FOREIGN KEY (`examReviewID`) REFERENCES `examReview`(`id`),
  CONSTRAINT `fk_qreview_toeic` FOREIGN KEY (`toeicQuestionID`) REFERENCES `toeicQuestion`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `custom_exam` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `user_id` INT NOT NULL,
  `exam_id` INT NOT NULL,
  CONSTRAINT `fk_custom_exam_user` FOREIGN KEY (`user_id`) REFERENCES `user`(`id`),
  CONSTRAINT `fk_custom_exam_exam` FOREIGN KEY (`exam_id`) REFERENCES `exam`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `PaymentTransaction` (
  `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
  `orderId` VARCHAR(100) NULL,
  `amount` BIGINT NULL,
  `currency` VARCHAR(20) NULL,
  `paymentMethod` VARCHAR(50) NULL,
  `bankCode` VARCHAR(50) NULL,
  `transactionCode` VARCHAR(100) NULL,
  `responseCode` VARCHAR(20) NULL,
  `transactionStatus` VARCHAR(50) NULL,
  `description` VARCHAR(255) NULL,
  `transactionDate` DATETIME NULL,
  `secureHash` VARCHAR(255) NULL,
  `user_id` INT NOT NULL,
  `startDate` DATETIME NULL,
  `endDate` DATETIME NULL,
  CONSTRAINT `fk_payment_user` FOREIGN KEY (`user_id`) REFERENCES `user`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `attemp_log` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `createAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `userId` INT NULL,
  `examId` INT NULL
) ENGINE=InnoDB;

CREATE TABLE `testResult` (
  `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
  `userID` INT NOT NULL,
  `examID` INT NOT NULL,
  `createAt` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `duration` INT NOT NULL,
  `listening` FLOAT NULL,
  `reading` FLOAT NULL,
  CONSTRAINT `fk_testresult_user` FOREIGN KEY (`userID`) REFERENCES `user`(`id`),
  CONSTRAINT `fk_testresult_exam` FOREIGN KEY (`examID`) REFERENCES `exam`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `visit_log` (
  `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
  `visitedAt` DATETIME DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB;

CREATE TABLE `password_reset_token` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `email` VARCHAR(255) NOT NULL,
  `token` VARCHAR(255) NOT NULL,
  `createdAt` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `used` TINYINT(1) NOT NULL DEFAULT 0
) ENGINE=InnoDB;

CREATE TABLE `bank_toeic_question` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `part` VARCHAR(50) NULL,
  `detail` TEXT NULL,
  `result` VARCHAR(20) NULL,
  `audio` VARCHAR(255) NULL,
  `clarify` VARCHAR(255) NULL,
  `isPublic` TINYINT(1) DEFAULT 1,
  `source_toeic_id` INT NOT NULL UNIQUE,
  `contributor_id` INT NULL,
  `createdAt` DATETIME DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT `fk_bank_toeic_contributor` FOREIGN KEY (`contributor_id`) REFERENCES `user`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `bank_group_question` (
  `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
  `part` VARCHAR(50) NULL,
  `content` TEXT NULL,
  `isPublic` TINYINT(1) DEFAULT 1,
  `source_group_id` INT NOT NULL UNIQUE,
  `createdAt` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `contributor_id` INT NULL,
  CONSTRAINT `fk_bank_group_contributor` FOREIGN KEY (`contributor_id`) REFERENCES `user`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `bank_group_child_question` (
  `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
  `indexNumber` INT NULL,
  `detail` TEXT NULL,
  `result` VARCHAR(20) NULL,
  `bank_group_id` BIGINT NULL,
  `clarify` VARCHAR(255) NULL,
  CONSTRAINT `fk_bank_child_group` FOREIGN KEY (`bank_group_id`) REFERENCES `bank_group_question`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `bank_toeic_option` (
  `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
  `mark` VARCHAR(10) NULL,
  `detail` TEXT NULL,
  `bank_question_id` INT NULL,
  `bank_child_question_id` BIGINT NULL,
  CONSTRAINT `fk_bank_option_question` FOREIGN KEY (`bank_question_id`) REFERENCES `bank_toeic_question`(`id`),
  CONSTRAINT `fk_bank_option_child` FOREIGN KEY (`bank_child_question_id`) REFERENCES `bank_group_child_question`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `bank_image` (
  `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
  `url` VARCHAR(255) NULL,
  `bank_question_id` INT NULL,
  CONSTRAINT `fk_bank_image_question` FOREIGN KEY (`bank_question_id`) REFERENCES `bank_toeic_question`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `bank_group_image` (
  `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
  `imageKey` VARCHAR(255) NULL,
  `bank_group_id` BIGINT NULL,
  CONSTRAINT `fk_bank_group_image` FOREIGN KEY (`bank_group_id`) REFERENCES `bank_group_question`(`id`)
) ENGINE=InnoDB;

CREATE TABLE `bank_group_audio` (
  `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
  `audioKey` VARCHAR(255) NULL,
  `bank_group_id` BIGINT NULL,
  CONSTRAINT `fk_bank_group_audio` FOREIGN KEY (`bank_group_id`) REFERENCES `bank_group_question`(`id`)
) ENGINE=InnoDB;
